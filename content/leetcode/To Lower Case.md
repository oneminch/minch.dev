## My Solution

### JavaScript

```js
/**
 * @param {string} str
 * @return {string}
 */

const toLowerCase = (str) => {
  let char_code = 0;
  let lower_case_arr = [];
  for (let i = 0; i < str.length; i++) {
    char_code = str.charCodeAt(i);
    if (char_code >= 65 && char_code <= 90) char_code += 32;
    lower_case_arr.push(char_code);
  }
  return String.fromCharCode(...lower_case_arr);
};
```
